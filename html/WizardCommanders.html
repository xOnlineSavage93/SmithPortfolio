public class UnitSpawner : MonoBehaviour
{
    public GameObject wizardPrefab;
    public Transform spawnPoint;

    void Start()
    {
        SpawnWizard();
    }

    void SpawnWizard()
    {
        Instantiate(wizardPrefab, spawnPoint.position, spawnPoint.rotation);
    }
}
